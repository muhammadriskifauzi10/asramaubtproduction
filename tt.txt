        function requestFormInputNohpEmail(e) {
            e.preventDefault();

            let nohp = $("#nohp").val().trim();
            let email = $("#email").val().trim();

            // Validasi kosong
            if (nohp === "" || email === "") {
                Swal.fire({
                    text: `Nomor HP dan Email wajib diisi!`,
                    icon: 'warning'
                });
                return;
            }

            // Regex Indonesia
            const regexHp = /^(?:\+628|628|08)[0-9]{8,12}$/;

            // Regex Email
            const regexEmail = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;

            // Validasi format HP
            if (!regexHp.test(nohp)) {
                Swal.fire({
                    text: `Format Nomor HP tidak valid! Gunakan format Indonesia.\nContoh: 081234567890 atau 6281234567890`,
                    icon: 'error'
                });
                return;
            }

            // Validasi format Email
            if (!regexEmail.test(email)) {
                Swal.fire({
                    text: `Format Email tidak valid! Contoh: nama@email.com`,
                    icon: 'error'
                });
                return;
            }

            $.ajax({
                url: "{{ route('penghuni.kirimotp') }}",
                method: "POST",
                data: {
                    nohp: nohp,
                    email: email,
                },
                beforeSend: function() {
                    $("#btn_otp").prop("disabled", true).html(`
                        <strong class="d-flex align-items-center gap-2">
                            <div class="spinner-border spinner-border-sm" role="status">
                                <span class="visually-hidden">Loading...</span>
                            </div>
                            Sedang mengirim OTP...
                        </strong>
                    `);
                },
                success: function(response) {
                    if (response.status === 200) {
                        // pindah layer
                        $("#layerverifikasi1").addClass("d-none");
                        $("#layerverifikasi2").removeClass("d-none");

                        Swal.fire({
                            title: "Success",
                            html: response.message,
                            icon: response.icon,
                            timer: 5000,
                            showConfirmButton: false
                        });
                    }
                },
                error: function() {
                    Swal.fire("Error", "Terjadi kesalahan!", "error");
                },
                complete: function() {
                    $("#btn_otp").prop("disabled", false).html(`
                        <strong>Kirim OTP<strong>
                    `);
                }
            });
        }

        function requestFormVerifikasiOTP(e) {
            e.preventDefault();

            let nohp = $("#nohp").val().trim();
            let email = $("#email").val().trim();

            let otp_nohp = $("#otp_nohp").val().trim();
            let otp_email = $("#otp_email").val().trim();

            // Validasi kosong
            if (otp_nohp === "" || email === "") {
                Swal.fire({
                    text: `Nomor OTP wajib diisi!`,
                    icon: 'warning'
                });
                return;
            }

            $.ajax({
                url: "{{ route('penghuni.verifikasiotp') }}",
                method: "POST",
                data: {
                    nohp: nohp,
                    email: email,
                    otp_nohp: otp_nohp,
                    otp_email: otp_email,
                },
                beforeSend: function() {
                    $("#btn_verifikasi").prop("disabled", true).html(`
                        <strong class="d-flex align-items-center gap-2">
                            <div class="spinner-border spinner-border-sm" role="status">
                                <span class="visually-hidden">Loading...</span>
                            </div>
                            Sedang memverifikasi OTP...
                        </strong>
                    `);
                },
                success: function(response) {
                    if (response.status === 200) {
                        // pindah layer
                        $("#layerverifikasi1").addClass("d-none");
                        $("#layerverifikasi2").removeClass("d-none");

                        Swal.fire({
                            title: "Success",
                            html: response.message,
                            icon: response.icon,
                            timer: 5000,
                            showConfirmButton: false
                        });

                        setTimeout(() => {
                            window.location.href = '{{ route('penghuni.gantipassword') }}'
                        }, 2000);
                    } else {
                        Swal.fire({
                            text: response.message,
                            icon: response.icon
                        });
                    }
                },
                error: function() {
                    Swal.fire("Error", "Terjadi kesalahan!", "error");
                },
                complete: function() {
                    $("#btn_verifikasi").prop("disabled", false).html(`
                        <strong>Verifikasi OTP</strong>
                    `);
                }
            });
        }
